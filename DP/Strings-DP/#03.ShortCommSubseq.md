### #03. Shortest Common Subsequence

```
Input:   str1 = "geek",  str2 = "eke"
Output: "geeke"
```

**C++ Code**
```cpp
int shortSubSeq(string& s1, string& s2) {
    int m = s1.size(), n = s2.size();
    vector<vector<int>> dp(m + 1, vector<int>(n + 1));

    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (s1[i - 1] == s2[j - 1])
                dp[i][j] = 1 + dp[i - 1][j - 1];
            else
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
        }
    }

    return m + n - dp[m][n];
}
```
