#### #03. Distinct Substrings Count

```
Input: abc
Output: 7  -> {‘a’, ‘b’, ‘c’, “ab”, “bc”, “abc”, “” }
```

**C++ Code**
```cpp
struct Node {
    Node* links[26];
    bool flag = false;

    bool containsKey(char ch) {
        return links[ch - 'a'] != NULL;
    }
    void put(char ch, Node* node) {
        links[ch - 'a'] = node;
    }
    Node* get(char ch) {
        return links[ch - 'a'];
    }
    void setEnd() {
        flag = true;
    }
    bool isEnd() {
        return flag;
    }
};

class Trie {
private:
    Node* root;
public:
    Trie() {
        root = new Node();
    }

    int getSubstringsCount(string &word) {
        int cnt = 0;
        for (int i = 0; i < word.size(); i++) {
            Node* node = root;
            for (int j = i; j < word.size(); j++) {
                char ch = word[j];
                if (!node->containsKey(ch)) {
                    node->put(ch, new Node());
                    cnt++;
                }
                node = node->get(ch);
            }
            node->setEnd();
        }
        return cnt + 1;
    }
};

int countDistinctSubstrings(string &word) {
    //    Write your code here.
    Trie trie;
    return trie.getSubstringsCount(word);
}
```
