### #03. Evaluate/Bracketize Expresson to be True, Possible?

```
Input: T|F&T^F
Output: True, Yes can be written as (T|F)&(T^F) or any else.
```

**Code**
```cpp
bool solve(string& s, int i, int j, bool reqTrue) {
    if (i > j)
        return false;
    
    //single element case
    if (i == j) {
        if (reqTrue)
            return s[i] == 'T';
        else
            return s[i] == 'F';
    }

    int ans = 0;
    
    //basically loops over symbols - (& | ^)
    for (int k = i + 1; k < j; k += 2) {
        int lt = solve(s, i, k - 1, true);
        int lf = solve(s, k + 1, j, false);
        int rt = solve(s, i, k - 1, true);
        int rf = solve(s, k + 1, j, false);

        //list all possible combinations
        if (s[k] == '|') {
            if (reqTrue)
                ans += lt * rt + lf * rt + lt * rf;
            else
                ans += lf * rf;
        }
        else if (s[k] == '&') {
            if (reqTrue)
                ans += lt * rt;
            else
                ans += lf * rf + lf * rt + lt * rf;
        }
        else if (s[k] == '^') {
            if (reqTrue)
                ans += lt * rf + lf * rt;
            else
                ans += lf * rf + lt * rt;
        }
    }

    return ans;
}

int main() {
    string s = "T|F&T^F";
    cout << solve(s, 0, s.size() - 1, true); //if question asks for 'True', set accordingly
    return 0;
}
```
