#### #02. Detect Cycle

**DFS**
```cpp
unordered_set<int> seen;
unordered_map<int, vector<int>> graph;

bool hasCycle(int start, int prev) {
    if (seen.count(start))
        return true;

    seen.insert(start);
    for (auto neighbour : graph[start])
        if (neighbour != prev && hasCycle(neighbour, start))
            return true;

    seen.erase(start);
    return false;
}
```

**BFS**
```cpp
unordered_map<int, vector<int>> graph;
unordered_set<int> seen;

bool hasCycle(int start) {
    queue<pair<int, int>> q;
    
    seen.insert(start);
    q.push({start, -1});

    while (!q.empty()) {
        int cur = q.front().first;
        int prev = q.front().second;
        q.pop();

        for (auto it : graph[cur]) {
            if (seen.count(it) && it != prev)
                return true;

            seen.insert(it);
            q.push({it, cur});
        }
    }

    return false;
}
```

**Kahn's Algorithm - BFS**
```cpp
void cycleCheck(vector<vector<int>>& graph, int n) {
    vector<int> indegree(n);
    queue<int> q;

    //count indegrees
    for (int i = 0; i < n; i++)
        for (auto it : graph[i])
            indegree[it]++;

    //add non-dependent (0 indegree) to queue
    for (int i = 0; i < n; i++)
        if (indegree[i] == 0)
            q.push(i);

    int cnt = 0;  //counts topological ordered visits
    vector<int> res;
    while (!q.empty()) {
        int node = q.front(); q.pop();
        res.push_back(node);
        cnt++;

        for (int neighbour : graph[node]) {
            indegree[neighbour]--;
            if (indegree[neighbour] == 0)
                q.push(neighbour);
        }
    }

    if (cnt == n)
        cout << "No cycle";
    else
        cout << "Cycle present";
}
```
